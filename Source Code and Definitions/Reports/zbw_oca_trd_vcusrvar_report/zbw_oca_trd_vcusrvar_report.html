SELECT
  a~pgmid     ,
  a~object    ,
  a~as4pos    ,
  a~obj_name  ,
  b~trkorr    ,
  c~as4text   ,
  b~trfunction,
  b~trstatus  ,
  b~strkorr   ,
  b~as4date   ,
  b~as4time   ,
  b~as4user
FROM
  e071 AS a
  INNER JOIN
    e070 AS b
    ON
      a~trkorr = b~trkorr
  LEFT JOIN
    e07t AS c
    ON
      a~trkorr = c~trkorr
INTO CORRESPONDING FIELDS OF TABLE @it_trdetails
WHERE
  b~as4date BETWEEN '10000101' AND '99991231'
  AND c~langu    = 'E'
  AND a~pgmid    = 'R3TR'
  AND a~object   = 'CDAT'
  AND a~obj_name = 'VCUSRVAR'.

SELECT * FROM swfeature INTO TABLE lt_swfeature.
SORT lt_swfeature BY mod_date.
READ TABLE lt_swfeature INTO wa_swfeature INDEX 1.
lv_systype = wa_swfeature-name.
CLEAR: lt_swfeature.

CALL FUNCTION 'ENQUEUE_E_TABLE'
  EXPORTING
    mode_rstable   = 'E'
    tabname        = lv_cust_table_name
  EXCEPTIONS
    foreign_lock   = 1
    system_failure = 2
    OTHERS         = 3.

DELETE FROM zbw_oca_trdvcusr.

LOOP AT it_trdetails WHERE sysname IS INITIAL.
  it_trdetails-sysname = lv_systype.
  it_trdetails-systemid = sy-sysid.
  MODIFY it_trdetails INDEX sy-tabix.
ENDLOOP.

MODIFY zbw_oca_trdvcusr FROM TABLE it_trdetails.

CALL FUNCTION 'DEQUEUE_E_TABLE'
  EXPORTING
    mode_rstable = 'E'
    tabname      = lv_cust_table_name.
